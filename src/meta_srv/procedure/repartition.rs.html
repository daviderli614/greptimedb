71    pub catalog_name: String,
72    pub schema_name: String,
73    pub table_name: String,
74    pub table_id: TableId,
75    pub plans: Vec<RepartitionPlanEntry>,

78impl PersistentContext {
79    pub fn new(
80        TableName {
81            catalog_name,
82            schema_name,
83            table_name,

85        table_id: TableId,
86    ) -> Self {

88            catalog_name,

98            CatalogLock::Read(&self.catalog_name).into(),

141    /// Retrieves the table route value for the given table id.
142    ///
143    /// Retry:
144    /// - Failed to retrieve the metadata of table.
145    ///
146    /// Abort:
147    /// - Table route not found.
148    pub async fn get_table_route_value(
149        &self,
150    ) -> Result<DeserializedValueWithBytes<TableRouteValue>> {
151        let table_id = self.persistent_ctx.table_id;
152        let table_route_value = self
153            .table_metadata_manager
154            .table_route_manager()
155            .table_route_storage()
156            .get_with_raw_bytes(table_id)
157            .await
158            .map_err(BoxedError::new)
159            .with_context(|_| error::RetryLaterWithSourceSnafu {
160                reason: format!("Failed to get table route for table: {}", table_id),
161            })?
162            .context(error::TableRouteNotFoundSnafu { table_id })?;
163
164        Ok(table_route_value)
165    }
166
167    /// Retrieves the table info value for the given table id.
168    ///
169    /// Retry:
170    /// - Failed to retrieve the metadata of table.
171    ///
172    /// Abort:
173    /// - Table info not found.
174    pub async fn get_table_info_value(&self) -> Result<TableInfoValue> {
175        let table_id = self.persistent_ctx.table_id;
176        let table_info_value = self
177            .table_metadata_manager
178            .table_info_manager()
179            .get(table_id)
180            .await
181            .map_err(BoxedError::new)
182            .with_context(|_| error::RetryLaterWithSourceSnafu {
183                reason: format!("Failed to get table info for table: {}", table_id),
184            })?
185            .context(error::TableInfoNotFoundSnafu { table_id })?
186            .into_inner();
187        Ok(table_info_value)
188    }
189
190    /// Updates the table route.
191    ///
192    /// Retry:
193    /// - Failed to retrieve the metadata of datanode table.
194    ///
195    /// Abort:
196    /// - Table route not found.
197    /// - Failed to update the table route.
198    pub async fn update_table_route(
199        &self,
200        current_table_route_value: &DeserializedValueWithBytes<TableRouteValue>,
201        new_region_routes: Vec<RegionRoute>,
202        new_region_wal_options: HashMap<RegionNumber, String>,
203    ) -> Result<()> {
204        let table_id = self.persistent_ctx.table_id;
205        if new_region_routes.is_empty() {
206            return error::UnexpectedSnafu {
207                violated: format!("new_region_routes is empty for table: {}", table_id),
208            }
209            .fail();
210        }
211        let datanode_id = new_region_routes
212            .first()
213            .unwrap()
214            .leader_peer
215            .as_ref()
216            .context(error::NoLeaderSnafu)?
217            .id;
218        let datanode_table_value =
219            get_datanode_table_value(&self.table_metadata_manager, table_id, datanode_id).await?;
220
221        let RegionInfo {
222            region_options,
223            region_wal_options,
224            ..
225        } = &datanode_table_value.region_info;
226
227        // Merge and validate the new region wal options.
228        let validated_region_wal_options =
229            crate::procedure::repartition::utils::merge_and_validate_region_wal_options(
230                region_wal_options,
231                new_region_wal_options,
232                &new_region_routes,
233                table_id,
234            )?;
235        info!(
236            "Updating table route for table: {}, new region routes: {:?}",
237            table_id, new_region_routes
238        );
239        self.table_metadata_manager
240            .update_table_route(
241                table_id,
242                datanode_table_value.region_info.clone(),
243                current_table_route_value,
244                new_region_routes,
245                region_options,
246                &validated_region_wal_options,
247            )
248            .await
249            .context(error::TableMetadataManagerSnafu)
250    }
251
252    /// Broadcasts the invalidate table cache message.
253    pub async fn invalidate_table_cache(&self) -> Result<()> {
254        let table_id = self.persistent_ctx.table_id;
255        let subject = format!(
256            "Invalidate table cache for repartition table, table: {}",
257            table_id,
258        );
259        let ctx = common_meta::cache_invalidator::Context {
260            subject: Some(subject),
261        };
262        let _ = self
263            .cache_invalidator
264            .invalidate(&ctx, &[CacheIdent::TableId(table_id)])
265            .await;
266        Ok(())
267    }

277#[async_trait::async_trait]
278#[typetag::serde(tag = "repartition_state")]
279pub(crate) trait State: Sync + Send + Debug {
280    fn name(&self) -> &'static str {
281        let type_name = std::any::type_name::<Self>();
282        // short name
283        type_name.split("::").last().unwrap_or(type_name)
284    }
285
286    /// Yields the next [State] and [Status].
287    async fn next(
288        &mut self,
289        ctx: &mut Context,
290        procedure_ctx: &ProcedureContext,
291    ) -> Result<(Box<dyn State>, Status)>;
292
293    fn as_any(&self) -> &dyn Any;
294}

296pub struct RepartitionProcedure {
297    state: Box<dyn State>,
298    context: Context,
299}

301#[derive(Debug, Serialize)]
302struct RepartitionData<'a> {
303    state: &'a dyn State,
304    persistent_ctx: &'a PersistentContext,
305}
306
307#[derive(Debug, Deserialize)]
308struct RepartitionDataOwned {
309    state: Box<dyn State>,
310    persistent_ctx: PersistentContext,
311}
312
313impl RepartitionProcedure {
314    const TYPE_NAME: &'static str = "metasrv-procedure::Repartition";
315
316    pub fn new(
317        from_exprs: Vec<PartitionExpr>,
318        to_exprs: Vec<PartitionExpr>,
319        context: Context,
320    ) -> Self {
321        let state = Box::new(RepartitionStart::new(from_exprs, to_exprs));
322
323        Self { state, context }
324    }
325
326    pub fn from_json<F>(json: &str, ctx_factory: F) -> ProcedureResult<Self>
327    where
328        F: FnOnce(PersistentContext) -> Context,
329    {
330        let RepartitionDataOwned {
331            state,
332            persistent_ctx,
333        } = serde_json::from_str(json).context(FromJsonSnafu)?;
334        let context = ctx_factory(persistent_ctx);
335
336        Ok(Self { state, context })
337    }
338}

340#[async_trait::async_trait]
341impl Procedure for RepartitionProcedure {
342    fn type_name(&self) -> &str {
343        Self::TYPE_NAME
344    }
345
346    async fn execute(&mut self, _ctx: &ProcedureContext) -> ProcedureResult<Status> {
347        let state = &mut self.state;
348        let state_name = state.name();
349        // Log state transition
350        common_telemetry::info!(
351            "Repartition procedure executing state: {}, table_id: {}",
352            state_name,
353            self.context.persistent_ctx.table_id
354        );
355        match state.next(&mut self.context, _ctx).await {
356            Ok((next, status)) => {
357                *state = next;
358                Ok(status)
359            }
360            Err(e) => {
361                if e.is_retryable() {
362                    Err(ProcedureError::retry_later(e))
363                } else {
364                    error!(
365                        e;
366                        "Repartition procedure failed, table id: {}",
367                        self.context.persistent_ctx.table_id,
368                    );
369                    Err(ProcedureError::external(e))
370                }
371            }
372        }
373    }
374
375    fn rollback_supported(&self) -> bool {
376        // TODO(weny): support rollback.
377        false
378    }
379
380    fn dump(&self) -> ProcedureResult<String> {
381        let data = RepartitionData {
382            state: self.state.as_ref(),
383            persistent_ctx: &self.context.persistent_ctx,
384        };
385        serde_json::to_string(&data).context(ToJsonSnafu)
386    }
387
388    fn lock_key(&self) -> LockKey {
389        LockKey::new(self.context.persistent_ctx.lock_key())
390    }
391
392    fn user_metadata(&self) -> Option<UserMetadata> {
393        // TODO(weny): support user metadata.
394        None
395    }
396}

411
412impl RepartitionProcedureFactory for DefaultRepartitionProcedureFactory {
413    fn create(
414        &self,
415        ddl_ctx: &DdlContext,
416        table_name: TableName,
417        table_id: TableId,
418        from_exprs: Vec<String>,
419        to_exprs: Vec<String>,
420    ) -> std::result::Result<BoxedProcedure, BoxedError> {
421        let persistent_ctx = PersistentContext::new(table_name, table_id);
422        let from_exprs = from_exprs
423            .iter()
424            .map(|e| {
425                PartitionExpr::from_json_str(e)
426                    .context(error::DeserializePartitionExprSnafu)?
427                    .context(error::EmptyPartitionExprSnafu)
428            })
429            .collect::<Result<Vec<_>>>()
430            .map_err(BoxedError::new)?;
431        let to_exprs = to_exprs
432            .iter()
433            .map(|e| {
434                PartitionExpr::from_json_str(e)
435                    .context(error::DeserializePartitionExprSnafu)?
436                    .context(error::EmptyPartitionExprSnafu)
437            })
438            .collect::<Result<Vec<_>>>()
439            .map_err(BoxedError::new)?;
440
441        let procedure = RepartitionProcedure::new(
442            from_exprs,
443            to_exprs,
444            Context::new(
445                ddl_ctx,
446                self.mailbox.clone(),
447                self.server_addr.clone(),
448                persistent_ctx,
449            ),
450        );
451
452        Ok(Box::new(procedure))
453    }
454
455    fn register_loaders(
456        &self,
457        ddl_ctx: &DdlContext,
458        procedure_manager: &ProcedureManagerRef,
459    ) -> std::result::Result<(), BoxedError> {
460        // Registers the repartition procedure loader.
461        let mailbox = self.mailbox.clone();
462        let server_addr = self.server_addr.clone();
463        let moved_ddl_ctx = ddl_ctx.clone();
464        procedure_manager
465            .register_loader(
466                RepartitionProcedure::TYPE_NAME,
467                Box::new(move |json| {
468                    let mailbox = mailbox.clone();
469                    let server_addr = server_addr.clone();
470                    let ddl_ctx = moved_ddl_ctx.clone();
471                    let factory = move |persistent_ctx| {
472                        Context::new(&ddl_ctx, mailbox, server_addr, persistent_ctx)
473                    };
474                    RepartitionProcedure::from_json(json, factory).map(|p| Box::new(p) as _)
475                }),
476            )
477            .map_err(BoxedError::new)?;
478
479        // Registers the repartition group procedure loader.
480        let mailbox = self.mailbox.clone();
481        let server_addr = self.server_addr.clone();
482        let moved_ddl_ctx = ddl_ctx.clone();
483        procedure_manager
484            .register_loader(
485                RepartitionGroupProcedure::TYPE_NAME,
486                Box::new(move |json| {
487                    let mailbox = mailbox.clone();
488                    let server_addr = server_addr.clone();
489                    let ddl_ctx = moved_ddl_ctx.clone();
490                    let factory = move |persistent_ctx| {
491                        RepartitionGroupContext::new(&ddl_ctx, mailbox, server_addr, persistent_ctx)
492                    };
493                    RepartitionGroupProcedure::from_json(json, factory).map(|p| Box::new(p) as _)
494                }),
495            )
496            .map_err(BoxedError::new)?;
497
498        Ok(())
499    }
