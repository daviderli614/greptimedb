32#[async_trait::async_trait]
33pub(crate) trait SchedulerCtx: Send + Sync {
34    async fn get_table_to_region_stats(&self) -> Result<HashMap<TableId, Vec<RegionStat>>>;
35
36    async fn get_table_route(
37        &self,
38        table_id: TableId,
39    ) -> Result<(TableId, PhysicalTableRouteValue)>;
40
41    async fn gc_regions(
42        &self,
43        region_ids: &[RegionId],
44        full_file_listing: bool,
45        timeout: Duration,
46    ) -> Result<GcReport>;
47}
48
49pub(crate) struct DefaultGcSchedulerCtx {
50    /// The metadata manager.
51    pub(crate) table_metadata_manager: TableMetadataManagerRef,
52    /// Procedure manager.
53    pub(crate) procedure_manager: ProcedureManagerRef,
54    /// For getting `RegionStats`.
55    pub(crate) meta_peer_client: MetaPeerClientRef,
56    /// The mailbox to send messages.
57    pub(crate) mailbox: MailboxRef,
58    /// The server address.
59    pub(crate) server_addr: String,
60}
61
62impl DefaultGcSchedulerCtx {
63    pub fn try_new(
64        table_metadata_manager: TableMetadataManagerRef,
65        procedure_manager: ProcedureManagerRef,
66        meta_peer_client: MetaPeerClientRef,
67        mailbox: MailboxRef,
68        server_addr: String,
69    ) -> Result<Self> {
70        Ok(Self {
71            table_metadata_manager,
72            procedure_manager,
73            meta_peer_client,
74            mailbox,
75            server_addr,
76        })
77    }
78}

80#[async_trait::async_trait]
81impl SchedulerCtx for DefaultGcSchedulerCtx {
82    async fn get_table_to_region_stats(&self) -> Result<HashMap<TableId, Vec<RegionStat>>> {
83        let dn_stats = self.meta_peer_client.get_all_dn_stat_kvs().await?;
84        let mut table_to_region_stats: HashMap<TableId, Vec<RegionStat>> = HashMap::new();
85        for (_dn_id, stats) in dn_stats {
86            let stats = stats.stats;
87
88            let Some(latest_stat) = stats.iter().max_by_key(|s| s.timestamp_millis).cloned() else {
89                continue;
90            };
91
92            for region_stat in latest_stat.region_stats {
93                table_to_region_stats
94                    .entry(region_stat.id.table_id())
95                    .or_default()
96                    .push(region_stat);
97            }
98        }
99        Ok(table_to_region_stats)
100    }
101
102    async fn get_table_route(
103        &self,
104        table_id: TableId,
105    ) -> Result<(TableId, PhysicalTableRouteValue)> {
106        self.table_metadata_manager
107            .table_route_manager()
108            .get_physical_table_route(table_id)
109            .await
110            .context(TableMetadataManagerSnafu)
111    }
112
113    async fn gc_regions(
114        &self,
115        region_ids: &[RegionId],
116        full_file_listing: bool,
117        timeout: Duration,
118    ) -> Result<GcReport> {
119        self.gc_regions_inner(region_ids, full_file_listing, timeout)
120            .await
121    }
122}
123
124impl DefaultGcSchedulerCtx {
125    async fn gc_regions_inner(
126        &self,
127        region_ids: &[RegionId],
128        full_file_listing: bool,
129        timeout: Duration,
130    ) -> Result<GcReport> {
131        debug!(
132            "Sending GC instruction for {} regions (full_file_listing: {})",
133            region_ids.len(),
134            full_file_listing
135        );
136
137        let procedure = BatchGcProcedure::new(
138            self.mailbox.clone(),
139            self.table_metadata_manager.clone(),
140            self.server_addr.clone(),
141            region_ids.to_vec(),
142            full_file_listing,
143            timeout,
144        );
145        let procedure_with_id = ProcedureWithId::with_random_id(Box::new(procedure));
146
147        let id = procedure_with_id.id;
148
149        let mut watcher = self
150            .procedure_manager
151            .submit(procedure_with_id)
152            .await
153            .context(error::SubmitProcedureSnafu)?;
154        let res = watcher::wait(&mut watcher)
155            .await
156            .context(error::WaitProcedureSnafu)?
157            .with_context(|| error::UnexpectedSnafu {
158                violated: format!(
159                    "GC procedure {id} successfully completed but no result returned"
160                ),
161            })?;
162
163        let gc_report = BatchGcProcedure::cast_result(res)?;
164
165        Ok(gc_report)
166    }
